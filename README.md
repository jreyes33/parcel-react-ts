```sh
npm init -y
git init
npm i -D @types/node @types/react @types/react-dom parcel-bundler typescript
echo $(npm bin)
$(npm bin)/parcel --help
$(npm bin)/parcel serve
$(npm bin)/parcel serve index.html
$(npm bin)/parcel index.html
npm help start
$(npm bin)/tsc --noEmit
ls -la
rm -rf .cache
npm start
npm i -D @types/node @types/parcel-env @types/react @types/react-dom parcel-bundler typescript
npm i -D @types/node@^12.0.0 @types/parcel-env @types/react @types/react-dom parcel-bundler typescript
npm run tsc
rg ";"
rm -rf dist
cd dist
python3 -m http.server
cd ..
npm run build
ls dist
du -sh dist
git add .
git diff --cached --stat
git diff --cached
git st
git commit -m "Initial commit"
```
